NEW endpoint
// Request body
{ identifier: "user@email.com" }

// Logic:
// 1. identifier not found in auth_users.email
// 2. Query auth_users WHERE email IS NULL
//    AND phone_verified = true
// This returns candidate orphaned accounts
// 3. Return existence of candidate

// Response A: no orphan found → create new account
{ resolution: 'new_account' }

// Response B: orphaned phone account exists
{ resolution: 'link_to_phone',
  hint: 'partial_name' } // first 2 chars of name for UI

// name.toLowerCase().trim() must start with
// or be a fuzzy match against auth_users.name
// (3+ chars, case-insensitive)

// On mismatch: error "Nombre no coincide"

Security note: The resolve-identifier-mismatch endpoint must NOT reveal whether a phone account exists without the email OTP first being verified. The hint (partial name) is only returned after Email OTP succeeds. This prevents account enumeration.

shared-auth.js — renderOTPScreen(type)
// Single function handles both cases:
function renderOTPScreen(type) {
  // type: 'phone' | 'email'
  const es = isEs();
  const title = type === 'phone'
    ? (es ? 'Verificar teléfono' : 'Verify phone')
    : (es ? 'Verificar correo'   : 'Verify email');

  const subtitle = type === 'phone'
    ? (es ? 'Enviamos un código SMS a '
          : 'We sent an SMS code to ')
    : (es ? 'Enviamos un código a '
          : 'We sent a code to ');

  // render modal with dynamic title + subtitle
  // identifier shown is uslIdentifier (current)
}

-- Verification tracking (from v1)
ALTER TABLE auth_users ADD COLUMN IF NOT EXISTS
  phone_verified BOOLEAN DEFAULT false;

ALTER TABLE auth_users ADD COLUMN IF NOT EXISTS
  email_verified BOOLEAN DEFAULT false;

ALTER TABLE auth_users ADD COLUMN IF NOT EXISTS
  first_identifier_type VARCHAR(10);
-- 'phone' | 'email'

-- Backfill existing phone-verified users
UPDATE auth_users
SET phone_verified = true,
    first_identifier_type = 'phone'
WHERE phone IS NOT NULL
  AND firebase_uid IS NOT NULL;

-- Backfill Google/email users
UPDATE auth_users
SET email_verified = true,
    first_identifier_type = 'email'
WHERE email IS NOT NULL
  AND auth_provider IN ('google', 'email')
  AND phone_verified = false;
